import org.jfugue.player.Player;
import org.jfugue.rhythm.Rhythm;

public void start() {
    Rhythm rhythm = new Rhythm()
        .addLayer("O..oO...O..oOO..")
        .addLayer("..S...S...S...S.")
        .addLayer("````````````````")
        .addLayer("...............+");
    new Player().play(rhythm.getPattern().repeat(2));
}
